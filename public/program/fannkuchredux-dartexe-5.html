import 'dart:isolate';
import 'dart:typed_data';
import 'dart:math';

Int32List nFactorials(int n) {
    final fact = Int32List(n+1);
    fact[0] = 1;
    for (int i = 1; i <= n; i++) {
        fact[i] = fact[i - 1] * i;
    }
    return fact;
}

void fannkuch(Request r) {
        final n = r.n;   
        final Int32List fact = r.fact;              

        final p = Int32List(n);
        final pp = Int32List(n);
        final count = Int32List(n);

        // first permutation
        for (var i = 0; i < n; i++) {
            p[i] = i;
        }
        int idx = r.idxMin;
        for (var i = n - 1; i > 0; i--) {
            final d = div(idx, fact[i]);
            count[i] = d;
            idx = idx % fact[i];

            for (var j = 0; j < n; j++) {
                pp[j] = p[j];
            }

            for (var j = 0; j <= i; j++) {
                if (j + d <= i) {
                    p[j] = pp[j + d];
                } else {
                    p[j] = pp[j + d - i - 1];
                }
            }
        }
        
        final reply = Reply();         

        idx = r.idxMin;
        for (var sign = true; ; sign = !sign) {
            var first = p[0];
            if (first != 0) {
                var flips = 1;
                if (p[first] != 0) {
                    for (var j = 1; j < n; j++) {
                        pp[j] = p[j];
                    }
                    var p0 = first;
                    while (true) {
                        flips++;
                        var i = 1;
                        var j = p0 - 1;
                        while (i < j) {
                            final t = pp[i];
                            pp[i] = pp[j];
                            pp[j] = t;
                            i++;
                            j--;
                        }
                        final t = pp[p0];
                        pp[p0] = p0;
                        p0 = t;
                        if (pp[p0] == 0) {
                            break;
                        }
                    }
                }
                
                if (reply.flips < flips) {
                    reply.flips = flips;
                }
                if (sign) {
                    reply.sum += flips;
                } else {
                    reply.sum -= flips;
                } 
                        
            }
            
            idx++;
            if (idx == r.idxMax) {
                break;
            }            
        
            // next permutation
            if (sign) {
                p[0] = p[1];
                p[1] = first;
            } else {
                final t = p[1];
                p[1] = p[2];
                p[2] = t;
                for (var k = 2; ; k++) {
                    count[k]++;
                    if (count[k] <= k) {
                        break;
                    }
                    count[k] = 0;
                    for (var j = 0; j <= k; j++) {
                        p[j] = p[j + 1];
                    }
                    p[k + 1] = first;
                    first = p[0];
                }
            }        
        }
        r.replyPort.send(reply);        
}

int div(int val, int by) {
    return (val - val % by) ~/ by;
}

void main(List<String> args) {
    final n = (args.length > 0) ? int.parse(args[0]) : 7;
    final fact = nFactorials(n);  
        
    const nchunks = 4;
    var chunkSize = div((fact[n] + nchunks - 1), nchunks);
    chunkSize += chunkSize % 2;
    var awaited = div((fact[n] + chunkSize - 1), chunkSize);
    ReceivePort _replyPort = ReceivePort();        
                  
    for (var i = 0; i < awaited; i++) {
        final idxMin = chunkSize * i;
        final idxMax = min(fact[n], idxMin + chunkSize);
        final chunk = Request(n,idxMin,idxMax,fact,_replyPort.sendPort);     
        Isolate.spawn(fannkuch,chunk);     
    }        
        
    var checkSum = 0, maxFlips = 0; 
          
    _replyPort.listen((dynamic stuff) {
        final reply = stuff as Reply;
        checkSum += reply.sum; 
        if (reply.flips > maxFlips){
            maxFlips = reply.flips;       
        }        
              
        awaited--;               
        if (awaited < 1){
            print("$checkSum\nPfannkuchen($n) = $maxFlips");                   
            _replyPort.close();          
        }         
    });                    
}

class Request {
    int n = 0, idxMin = 0, idxMax = 0;   
    Int32List fact;     
    SendPort replyPort;       
    Request(this.n,this.idxMin,this.idxMax,this.fact,this.replyPort);
}

class Reply {
    int sum = 0, flips = 0;
}
