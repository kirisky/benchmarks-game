do {
    $str = trim(fgets(STDIN));
    if( $str && $str[0] == '>' ){
        # if we're on a comment line, print the previous seq and move on
		if($seq != ''){
		   echo wordwrap( strrev( strtr(strtoupper($seq),
			  'CGATMKRYVBHD', 'GCTAKMYRBVDH') ), 60, "\n", true ), "\n";
		   $seq = '';
		}
        echo $str, "\n";
    }else{
        # otherwise, just append to the sequence
        $seq .= $str;
    }
} while( !feof(STDIN) );
if($seq != ''){
	echo wordwrap( strrev( strtr(strtoupper($seq),
	   'CGATMKRYVBHD', 'GCTAKMYRBVDH') ), 60, "\n", true ), "\n";
